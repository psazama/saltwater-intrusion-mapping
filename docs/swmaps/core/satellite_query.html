 26def query_satellite_items(
 27    mission: str = "sentinel-2",
 28    bbox: list[float] | None = None,
 29    date_range: str | None = None,
 30    max_items: int | None = None,
 31    debug: bool = False,
 32) -> tuple[list[pystac.Item], dict[str, str]]:
 33    """Query the AWS Earth Search STAC API for imagery matching the criteria.
 34
 35    Args:
 36        mission (str): Mission slug understood by :func:`get_mission`.
 37        bbox (list[float] | None): Bounding box ``[minx, miny, maxx, maxy]``
 38            in WGS84 coordinates.
 39        date_range (str | None): ISO-8601 date or range string.
 40        max_items (int | None): Maximum number of STAC items to fetch.
 41        debug (bool): If ``True``, print identifiers for the retrieved items.
 42
 43    Returns:
 44        tuple[list[pystac.Item], dict[str, str]]: Matching STAC items and the
 45        band mapping for the mission.
 46
 47    Raises:
 48        ValueError: When no matching items are found.
 49    """
 50
 51    mission_specs = get_mission(mission)
 52    bands = mission_specs["bands"]
 53    collection = mission_specs["collection"]
 54    query_filter = mission_specs["query_filter"]
 55
 56    catalog = Client.open("https://earth-search.aws.element84.com/v1")
 57    search = catalog.search(
 58        collections=[collection],
 59        bbox=bbox,
 60        datetime=date_range,
 61        query=query_filter,
 62        max_items=max_items,
 63    )
 64
 65    items = list(search.items())
 66    if not items:
 67        raise ValueError("No items found for your search.")
 68
 69    if debug:
 70        for item in items:
 71            print(f"Found item: {item.id}")
 72
 73    return items, bands

 76def download_satellite_bands_from_item(
 77    item: pystac.Item,
 78    bands: dict[str, str],
 79    to_disk: bool = True,
 80    data_dir: str | Path | None = None,
 81    debug: bool = False,
 82    mission: str | None = None,
 83    downsample_to_landsat_res: bool = False,
 84    target_resolution: float = 30,
 85) -> list[tuple[int, np.ndarray, Affine, CRS]]:
 86    """Download selected bands for a STAC item.
 87
 88    Args:
 89        item (pystac.Item): STAC item describing the asset locations.
 90        bands (dict[str, str]): Mapping of STAC asset keys to file suffixes.
 91        to_disk (bool): When ``True``, persist downloaded bands to ``data_dir``.
 92        data_dir (str | Path | None): Directory for saved rasters.
 93        debug (bool): If ``True``, emit debug logging for each band.
 94        mission (str | None): Mission slug, used to determine resampling.
 95        downsample_to_landsat_res (bool): If ``True`` and mission is
 96            Sentinel-2, resample to Landsat resolution.
 97        target_resolution (float): Target resolution used when resampling.
 98
 99    Returns:
100        list[tuple[int, np.ndarray, Affine, CRS]]: Tuples containing the
101        one-based band index, data array, transform, and CRS for each band.
102    """
103
104    session = AWSSession(requester_pays=True)
105    data_dir = Path(data_dir) if data_dir else data_path()
106    if to_disk:
107        data_dir.mkdir(parents=True, exist_ok=True)
108    out_path = None
109
110    def _fetch_one(k_n):
111        """Download a single band asset and replace nodata values with ``NaN``.
112
113        Args:
114            k_n (tuple[str, str]): Pair of the STAC asset key and the desired
115                filename stem for the downloaded band.
116
117        Returns:
118            tuple[str, str, np.ndarray, Affine, CRS]: Asset key, friendly band
119            name, array data, affine transform, and CRS captured from the
120            source dataset.
121        """
122        key, name = k_n
123        href = item.assets[key].href
124        with rasterio.Env(session):
125            with rasterio.open(href) as src:
126                arr = src.read(1).astype(np.float32)
127                nod = src.nodata or 0
128                arr[arr == nod] = np.nan
129                return key, name, arr, src.transform, src.crs
130
131    band_order = list(bands.keys())
132    band_data: list[tuple[int, np.ndarray, Affine, str]] = []
133
134    with ThreadPoolExecutor(max_workers=len(bands)) as pool:
135        for key, name, data, trf, crs in pool.map(_fetch_one, bands.items()):
136            if downsample_to_landsat_res and mission == "sentinel-2":
137                data, trf = downsample_to_landsat(data, trf, crs, target_resolution)
138
139            band_index = band_order.index(key) + 1
140            band_data.append((band_index, data, trf, crs))
141
142            if to_disk:
143                out_path = data_dir / f"{item.id}_{name}.tif"
144                profile = {
145                    "driver": "GTiff",
146                    "dtype": "float32",
147                    "nodata": np.nan,
148                    "crs": crs,
149                    "transform": trf,
150                    "height": data.shape[0],
151                    "width": data.shape[1],
152                }
153                with rasterio.open(out_path, "w", **profile, BIGTIFF="YES") as dst:
154                    dst.write(data, 1)
155
156            if debug:
157                nan_pct = np.isnan(data).sum() / data.size
158                logging.debug(f"{item.id}  {name}: {data.shape}  NaNs={nan_pct:.2%}")
159
160    return band_data

163def _stack_bands(
164    item: pystac.Item,
165    bands: dict[str, str],
166) -> tuple[np.ndarray, Affine, rasterio.crs.CRS]:
167    """Fetch and stack all requested bands for a STAC item into a single array.
168
169    Args:
170        item (pystac.Item): STAC item holding band assets.
171        bands (dict[str, str]): Mapping of band identifiers to asset keys.
172
173    Returns:
174        tuple[np.ndarray, Affine, rasterio.crs.CRS]: Stacked band array,
175        affine transform, and CRS associated with the imagery.
176    """
177    session = AWSSession(requester_pays=True)
178    arrays = []
179    transforms = None
180    crs = None
181    for key in bands.keys():
182        href = item.assets[key].href
183        with rasterio.Env(session), rasterio.open(href) as src:
184            arr = src.read(1).astype(np.float32)
185            if transforms is None:
186                transforms, crs = src.transform, src.crs
187            arrays.append(arr)
188    return np.stack(arrays), transforms, crs

191def find_satellite_coverage(
192    df: pd.DataFrame,
193    missions: list[str] = ["sentinel-2", "landsat-5", "landsat-7"],
194    buffer_km: float = 10,  # increased default buffer
195    days_before: int = 7,  # new param: extend temporal window
196    days_after: int = 7,
197) -> pd.DataFrame:
198    """Annotate salinity observations with missions that have nearby imagery.
199       Rows within the same buffer cluster share the same `covered_by` result.
200
201    Args:
202        df (pandas.DataFrame): Observation table containing ``latitude``,
203            ``longitude``, and ``date`` columns.
204        missions (list[str]): Mission slugs to consider.
205        buffer_km (float): Spatial buffer radius around each observation.
206        days_before (int): Days before observation date to include in query.
207        days_after (int): Days after observation date to include in query.
208
209    Returns:
210        pandas.DataFrame: Input frame with an added ``covered_by`` column
211        listing missions that provide imagery.
212    """
213    buffer_deg = buffer_km / 111.0
214
215    # Step 1: make GeoDataFrame
216    gdf = gpd.GeoDataFrame(
217        df.copy(),
218        geometry=gpd.points_from_xy(df["longitude"], df["latitude"]),
219        crs="EPSG:4326",
220    )
221
222    # Step 2: buffer each point (in projected CRS, then back to WGS84)
223    gdf_proj = gdf.to_crs("EPSG:3857")  # Web Mercator (meters)
224    gdf_proj["buffer"] = gdf_proj.geometry.buffer(buffer_km * 1000)
225    gdf["buffer"] = gdf_proj.to_crs("EPSG:4326")["buffer"]
226
227    # Step 3: dissolve overlapping buffers into clusters
228    dissolved = (
229        gdf.set_geometry("buffer")
230        .dissolve()
231        .explode(index_parts=True)
232        .reset_index(drop=True)
233    )
234
235    # Step 4: assign cluster ids to each row
236    cluster_ids = []
237    for pt in tqdm(gdf.geometry, desc="Assigning clusters"):
238        cid = next(i for i, poly in enumerate(dissolved.geometry) if pt.within(poly))
239        cluster_ids.append(cid)
240    gdf["cluster_id"] = cluster_ids
241
242    # Step 5: query per cluster
243    cluster_results: dict[int, list[str]] = {}
244    for cid, cluster_df in tqdm(
245        gdf.groupby("cluster_id"), total=len(dissolved), desc="Querying STAC"
246    ):
247        rep = cluster_df.iloc[0]  # representative row
248        lat, lon, date = rep["latitude"], rep["longitude"], rep["date"]
249
250        try:
251            # extended date range
252            dt = datetime.strptime(date, "%Y-%m-%d")
253            date_range = (
254                f"{(dt - timedelta(days=days_before)).date()}/"
255                f"{(dt + timedelta(days=days_after)).date()}"
256            )
257            bbox = [
258                lon - buffer_deg,
259                lat - buffer_deg,
260                lon + buffer_deg,
261                lat + buffer_deg,
262            ]
263
264            covered_by = []
265            for mission in missions:
266                try:
267                    mission_info = get_mission(mission)
268                    start_date, end_date = mission_info["valid_date_range"]
269                    if start_date and dt < datetime.strptime(start_date, "%Y-%m-%d"):
270                        continue
271                    if end_date and dt > datetime.strptime(end_date, "%Y-%m-%d"):
272                        continue
273
274                    items, _ = query_satellite_items(
275                        mission=mission,
276                        bbox=bbox,
277                        date_range=date_range,
278                        max_items=3,  # fetch more than 1 for safety
279                    )
280                    if items:
281                        covered_by.append(mission)
282                except Exception:
283                    pass
284
285            cluster_results[cid] = covered_by
286
287        except Exception as e:
288            print(f"[ERROR] Cluster {cid} failed: {e}")
289            cluster_results[cid] = []
290
291    # Step 6: map cluster results back to each row
292    gdf["covered_by"] = gdf["cluster_id"].map(cluster_results)
293
294    return pd.DataFrame(gdf.drop(columns=["geometry", "buffer"]))

297def download_matching_images(
298    df: pd.DataFrame,
299    missions: list[str] = ["sentinel-2", "landsat-5", "landsat-7"],
300    buffer_km: float = 0.1,
301    output_dir: str | Path | None = None,
302) -> pd.DataFrame:
303    """Download imagery for each observation's matched missions.
304       Deduplicates downloads by checking if a lat/lon is already covered by a previously
305       downloaded STAC item (based on bbox containment).
306
307    Args:
308        df (pandas.DataFrame): Observation table with ``covered_by`` details.
309        missions (list[str]): Mission slugs considered for download.
310        buffer_km (float): Buffer radius to define the query bounding box.
311        output_dir (str | Path | None): Destination directory for downloads.
312
313    Returns:
314        pandas.DataFrame: DataFrame with an additional ``downloaded_files``
315        column listing downloaded multiband rasters per observation.
316    """
317    output_dir = Path(output_dir) if output_dir else data_path("matched_downloads")
318    output_dir.mkdir(parents=True, exist_ok=True)
319
320    downloaded_paths = []
321    seen_items: dict[str, dict] = (
322        {}
323    )  # item.id -> {"path": path, "bbox": bbox, "mission": mission}
324
325    for _, row in tqdm(df.iterrows(), total=len(df)):
326        lat, lon, date = row["latitude"], row["longitude"], row["date"]
327        dt = datetime.strptime(date, "%Y-%m-%d")
328        date_range = (
329            f"{(dt - timedelta(days=1)).date()}/{(dt + timedelta(days=1)).date()}"
330        )
331        buffer_deg = buffer_km / 111.0
332        bbox = [lon - buffer_deg, lat - buffer_deg, lon + buffer_deg, lat + buffer_deg]
333        point = Point(lon, lat)
334
335        row_downloads = []
336
337        for mission in row.get("covered_by", []):
338            try:
339                # --- Deduplication check ---
340                reused_path = None
341                for meta in seen_items.values():
342                    if meta["mission"] == mission and point.within(box(*meta["bbox"])):
343                        reused_path = meta["path"]
344                        break
345                if reused_path:
346                    row_downloads.append(reused_path)
347                    continue
348
349                # Query new item
350                items, bands = query_satellite_items(
351                    mission=mission, bbox=bbox, date_range=date_range, max_items=1
352                )
353                if not items:
354                    continue
355
356                item = items[0]
357                item_bbox = item.bbox
358                download_dir = output_dir / mission / item.id
359                download_dir.mkdir(parents=True, exist_ok=True)
360
361                multi_band_path = download_dir / f"{item.id}_multiband.tif"
362                if multi_band_path.exists():
363                    print(f"[SKIP] Already exists: {multi_band_path}")
364                    row_downloads.append(str(multi_band_path))
365                    seen_items[item.id] = {
366                        "path": str(multi_band_path),
367                        "bbox": item_bbox,
368                        "mission": mission,
369                    }
370                    continue
371
372                # Download each band
373                download_satellite_bands_from_item(
374                    item,
375                    bands,
376                    to_disk=True,
377                    data_dir=download_dir,
378                    mission=mission,
379                    downsample_to_landsat_res=(mission == "sentinel-2"),
380                )
381
382                # Stack into multiband GeoTIFF
383                band_arrays = []
384                first_profile = None
385                for band_key in bands.values():
386                    band_path = download_dir / f"{item.id}_{band_key}.tif"
387                    if not band_path.exists():
388                        raise FileNotFoundError(f"Expected band missing: {band_path}")
389                    with rasterio.open(band_path) as src:
390                        data = src.read(1)
391                        band_arrays.append(data)
392                        if first_profile is None:
393                            first_profile = src.profile.copy()
394
395                # --- Defensive checks ---
396                if not band_arrays:
397                    raise ValueError(
398                        f"No bands were read for {mission} {item.id}. "
399                        f"Band list: {list(bands.values())}"
400                    )
401                if first_profile is None:
402                    raise ValueError(
403                        f"Could not initialize raster profile for {mission} {item.id}"
404                    )
405
406                band_count = int(len(band_arrays))
407                if band_count < 1:
408                    raise ValueError(
409                        f"Invalid band count ({band_count}) for {mission} {item.id}"
410                    )
411
412                first_profile.update(count=band_count)
413
414                with rasterio.open(multi_band_path, "w", **first_profile) as dst:
415                    for i, band_array in enumerate(band_arrays, start=1):
416                        dst.write(band_array, i)
417
418                row_downloads.append(str(multi_band_path))
419                seen_items[item.id] = {
420                    "path": str(multi_band_path),
421                    "bbox": item_bbox,
422                    "mission": mission,
423                }
424
425            except Exception as e:
426                print(
427                    f"[ERROR] Could not download/combine {mission} for {lat},{lon} on {date}: {e}"
428                )
429                raise e
430
431        downloaded_paths.append(row_downloads)
432
433    df = df.copy()
434    df["downloaded_files"] = downloaded_paths
435    return df

208        days_after (int): Days after observation date to include in query.
209
210    Returns:
211        pandas.DataFrame: Input frame with an added ``covered_by`` column
212        listing missions that provide imagery.
213    """
214    buffer_deg = buffer_km / 111.0
215
216    # Step 1: make GeoDataFrame
217    gdf = gpd.GeoDataFrame(
218        df.copy(),
219        geometry=gpd.points_from_xy(df["longitude"], df["latitude"]),
220        crs="EPSG:4326",
221    )
222
223    # Step 2: buffer each point (in projected CRS, then back to WGS84)
224    gdf_proj = gdf.to_crs("EPSG:3857")  # Web Mercator (meters)
225    gdf_proj["buffer"] = gdf_proj.geometry.buffer(buffer_km * 1000)
226    gdf["buffer"] = gdf_proj.to_crs("EPSG:4326")["buffer"]
227
228    # Step 3: dissolve overlapping buffers into clusters
229    dissolved = (
230        gdf.set_geometry("buffer")
231        .dissolve()
232        .explode(index_parts=True)
233        .reset_index(drop=True)
234    )
235
236    # Step 4: assign cluster ids to each row
237    cluster_ids = []
238    for pt in tqdm(gdf.geometry, desc="Assigning clusters"):
239        cid = next(i for i, poly in enumerate(dissolved.geometry) if pt.within(poly))
240        cluster_ids.append(cid)
241    gdf["cluster_id"] = cluster_ids
242
243    # Step 5: query per cluster
244    cluster_results: dict[int, list[str]] = {}
245    for cid, cluster_df in tqdm(
246        gdf.groupby("cluster_id"), total=len(dissolved), desc="Querying STAC"
247    ):
248        rep = cluster_df.iloc[0]  # representative row
249        lat, lon, date = rep["latitude"], rep["longitude"], rep["date"]
250
251        try:
252            # extended date range
253            dt = datetime.strptime(date, "%Y-%m-%d")
254            date_range = (
255                f"{(dt - timedelta(days=days_before)).date()}/"
256                f"{(dt + timedelta(days=days_after)).date()}"
257            )
258            bbox = [
259                lon - buffer_deg,
260                lat - buffer_deg,
261                lon + buffer_deg,
262                lat + buffer_deg,
263            ]
264
265            covered_by = []
266            for mission in missions:
267                try:
268                    mission_info = get_mission(mission)
269                    start_date, end_date = mission_info["valid_date_range"]
270                    if start_date and dt < datetime.strptime(start_date, "%Y-%m-%d"):
271                        continue
272                    if end_date and dt > datetime.strptime(end_date, "%Y-%m-%d"):
273                        continue
274
275                    items, _ = query_satellite_items(
276                        mission=mission,
277                        bbox=bbox,
278                        date_range=date_range,
279                        max_items=3,  # fetch more than 1 for safety
280                    )
281                    if items:
282                        covered_by.append(mission)
283                except Exception:
284                    pass
285
286            cluster_results[cid] = covered_by
287
288        except Exception as e:
289            print(f"[ERROR] Cluster {cid} failed: {e}")
290            cluster_results[cid] = []
291
292    # Step 6: map cluster results back to each row
293    gdf["covered_by"] = gdf["cluster_id"].map(cluster_results)
294
295    return pd.DataFrame(gdf.drop(columns=["geometry", "buffer"]))

def download_matching_images( df: pandas.core.frame.DataFrame, missions: list[str] = ['sentinel-2', 'landsat-5', 'landsat-7'], buffer_km: float = 0.1, output_dir: str | pathlib._local.Path | None = None) -> pandas.core.frame.DataFrame: 
298def download_matching_images(
299    df: pd.DataFrame,
300    missions: list[str] = ["sentinel-2", "landsat-5", "landsat-7"],
301    buffer_km: float = 0.1,
302    output_dir: str | Path | None = None,
303) -> pd.DataFrame:
304    """Download imagery for each observation's matched missions.
305       Deduplicates downloads by checking if a lat/lon is already covered by a previously
306       downloaded STAC item (based on bbox containment).
307
308    Args:
309        df (pandas.DataFrame): Observation table with ``covered_by`` details.
310        missions (list[str]): Mission slugs considered for download.
311        buffer_km (float): Buffer radius to define the query bounding box.
312        output_dir (str | Path | None): Destination directory for downloads.
313
314    Returns:
315        pandas.DataFrame: DataFrame with an additional ``downloaded_files``
316        column listing downloaded multiband rasters per observation.
317    """
318    output_dir = Path(output_dir) if output_dir else data_path("matched_downloads")
319    output_dir.mkdir(parents=True, exist_ok=True)
320
321    downloaded_paths = []
322    seen_items: dict[str, dict] = (
323        {}
324    )  # item.id -> {"path": path, "bbox": bbox, "mission": mission}
325
326    for _, row in tqdm(df.iterrows(), total=len(df)):
327        lat, lon, date = row["latitude"], row["longitude"], row["date"]
328        dt = datetime.strptime(date, "%Y-%m-%d")
329        date_range = (
330            f"{(dt - timedelta(days=1)).date()}/{(dt + timedelta(days=1)).date()}"
331        )
332        buffer_deg = buffer_km / 111.0
333        bbox = [lon - buffer_deg, lat - buffer_deg, lon + buffer_deg, lat + buffer_deg]
334        point = Point(lon, lat)
335
336        row_downloads = []
337
338        for mission in row.get("covered_by", []):
339            try:
340                # --- Deduplication check ---
341                reused_path = None
342                for meta in seen_items.values():
343                    if meta["mission"] == mission and point.within(box(*meta["bbox"])):
344                        reused_path = meta["path"]
345                        break
346                if reused_path:
347                    row_downloads.append(reused_path)
348                    continue
349
350                # Query new item
351                items, bands = query_satellite_items(
352                    mission=mission, bbox=bbox, date_range=date_range, max_items=1
353                )
354                if not items:
355                    continue
356
357                item = items[0]
358                item_bbox = item.bbox
359                download_dir = output_dir / mission / item.id
360                download_dir.mkdir(parents=True, exist_ok=True)
361
362                multi_band_path = download_dir / f"{item.id}_multiband.tif"
363                if multi_band_path.exists():
364                    print(f"[SKIP] Already exists: {multi_band_path}")
365                    row_downloads.append(str(multi_band_path))
366                    seen_items[item.id] = {
367                        "path": str(multi_band_path),
368                        "bbox": item_bbox,
369                        "mission": mission,
370                    }
371                    continue
372
373                # Download each band
374                download_satellite_bands_from_item(
375                    item,
376                    bands,
377                    to_disk=True,
378                    data_dir=download_dir,
379                    mission=mission,
380                    downsample_to_landsat_res=(mission == "sentinel-2"),
381                )
382
383                # Stack into multiband GeoTIFF
384                band_arrays = []
385                first_profile = None
386                for band_key in bands.values():
387                    band_path = download_dir / f"{item.id}_{band_key}.tif"
388                    if not band_path.exists():
389                        raise FileNotFoundError(f"Expected band missing: {band_path}")
390                    with rasterio.open(band_path) as src:
391                        data = src.read(1)
392                        band_arrays.append(data)
393                        if first_profile is None:
394                            first_profile = src.profile.copy()
395
396                # --- Defensive checks ---
397                if not band_arrays:
398                    raise ValueError(
399                        f"No bands were read for {mission} {item.id}. "
400                        f"Band list: {list(bands.values())}"
401                    )
402                if first_profile is None:
403                    raise ValueError(
404                        f"Could not initialize raster profile for {mission} {item.id}"
405                    )
406
407                band_count = int(len(band_arrays))
408                if band_count < 1:
409                    raise ValueError(
410                        f"Invalid band count ({band_count}) for {mission} {item.id}"
411                    )
412
413                first_profile.update(count=band_count)
414
415                with rasterio.open(multi_band_path, "w", **first_profile) as dst:
416                    for i, band_array in enumerate(band_arrays, start=1):
417                        dst.write(band_array, i)
418
419                row_downloads.append(str(multi_band_path))
420                seen_items[item.id] = {
421                    "path": str(multi_band_path),
422                    "bbox": item_bbox,
423                    "mission": mission,
424                }
425
426            except Exception as e:
427                print(
428                    f"[ERROR] Could not download/combine {mission} for {lat},{lon} on {date}: {e}"
429                )
430                raise e
431
432        downloaded_paths.append(row_downloads)
433
434    df = df.copy()
435    df["downloaded_files"] = downloaded_paths
436    return df
